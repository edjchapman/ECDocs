```python filename="secret_key_generator.py" showLineNumbers copy
import random

def generate_secret_key(length: int = 50, excluded_chars: list = None) -> str:
    """
    Generates a random 50 character string of ascii letters, digits, and
    punctuation which can be used as a SECRET_KEY setting value.
    """
    seed = [
        c
        for c in (
            "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789!@#$^&*(-_=+)"
        )
        if c not in (excluded_chars or [])
    ]
    return "".join(
        [
            random.SystemRandom().choice(seed)
            for _ in range(length)
        ]
    )
```
